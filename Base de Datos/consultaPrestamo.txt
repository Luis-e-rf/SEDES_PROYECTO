Select td.nombre_tipo_documento, p.documento_Persona, p.Nombre_Persona, p.Apellido_Persona, p.Direccion_Persona, p.Telefono_Persona,p.Email,f.numero_ficha,pr.nombre_programa,af.id_aula, e.Nombre_elemento, es.Nombre_Estado
                from persona p, tipo_persona tp,tipo_documento td,ficha f, programa pr, persona_ficha pf, aula_ficha af,prestamos pre, elementos e, estado es
                where p.documento_Persona= 'documento_Persona' and p.id_tipo_documento=td.id_tipo_documento and p.Id_Tipo_Persona=tp.id_Tipo_Persona and p.documento_Persona=pf.documento_Persona and f.numero_ficha=pf.numero_ficha and  pr.id_programa=f.id_programa and tp.Nombre_Tipo_Persona='Aprendiz' and af.numero_ficha=f.numero_ficha and p.documento_Persona=pre.documento_Persona and es.Nombre_Estado=e.Id_Estado  order by  f.numero_ficha,p.Nombre_Persona ASC ;
